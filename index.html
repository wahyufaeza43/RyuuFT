Pasal 30"], "ans": "B"}, ] class Game: def __init__(self): self.reset() def reset(self): self.current_q = 0 self.score = 0 self.lives = 3 self.game_over = False self.timer = 1500 # 15 detik (60 FPS * 25) self.max_timer = 1500 def draw_text(self, text, font, color, x, y, center=False): img = font.render(text, True, color) rect = img.get_rect() if center: rect.center = (x, y) else: rect.topleft = (x, y) screen.blit(img, rect) def update(self): if not self.game_over: self.timer -= 2 # Kecepatan waktu berkurang if self.timer <= 0: self.lives -= 1 self.next_question() def next_question(self): if self.current_q < len(questions) - 1 and self.lives > 0: self.current_q += 1 self.timer = self.max_timer else: self.game_over = True def draw(self): screen.fill(BG_COLOR) # Header UI pygame.draw.rect(screen, PRIMARY, (0, 0, WIDTH, 100)) self.draw_text(f"SCORE: {self.score}", font_ui, WHITE := (255,255,255), 40, 35) self.draw_text(f"LIVES: {'❤️' * self.lives}", font_ui, DANGER, WIDTH -
